import javax.swing.*;
import javax.swing.table.DefaultTableModel;
import java.awt.*;
import java.awt.event.ActionEvent;
import java.awt.event.ActionListener;
import java.util.ArrayList;
import java.util.List;


class MenuItem{
private String name;
private double price;
public MenuItem(String name,double price){
this.name=name;
this.price=price;
}
public String getName(){
return name;
}
public double getPrice(){
return price;
}
}


class OrderItem{
private MenuItem item;
private int quantity;
public OrderItem(MenuItem item,int quantity){
this.item=item;
this.quantity=quantity;
}
public MenuItem getItem(){
return item;
}
public int getQuantity(){
return quantity;
}
public double getTotalPrice(){
return item.getPrice()*quantity;
}
}


class RestaurantBillingSystem{
private List<OrderItem> orderedItems;
public RestaurantBillingSystem(){
orderedItems=new ArrayList<>();
}
public void addOrderItem(String name,double price,int quantity){
MenuItem item=new MenuItem(name,price);
orderedItems.add(new OrderItem(item,quantity));
}
public double calculateTotalBill(){
double total=0.0;
for (OrderItem orderItem : orderedItems){
total+=orderItem.getTotalPrice();
}
return total;
}
public List<OrderItem> getOrderedItems(){
return orderedItems;
}
}


public class RestaurantBillingSystemGUI{
private static RestaurantBillingSystem billingSystem=new RestaurantBillingSystem();
private static JFrame frame;
private static DefaultTableModel tableModel;
private static JTextField nameField,priceField,quantityField,totalField;
public static void main(String[] args){
frame=new JFrame("Restaurant Billing System");
frame.setSize(600,450);
frame.setDefaultCloseOperation(JFrame.EXIT_ON_CLOSE);
frame.setLayout(new BorderLayout());
JPanel inputPanel=new JPanel();
inputPanel.setLayout(new GridLayout(4,2));
inputPanel.add(new JLabel("Item Name:"));
nameField=new JTextField();
inputPanel.add(nameField);
inputPanel.add(new JLabel("Item Price:"));
priceField=new JTextField();
inputPanel.add(priceField);
inputPanel.add(new JLabel("Quantity:"));
quantityField=new JTextField();
inputPanel.add(quantityField);
JButton addButton=new JButton("Add to Order");
inputPanel.add(addButton);
frame.add(inputPanel,BorderLayout.NORTH);
String[] columnNames={"Item Name","Price","Quantity","Total Price"};
tableModel=new DefaultTableModel(columnNames,0);
JTable orderTable=new JTable(tableModel);
JScrollPane scrollPane=new JScrollPane(orderTable);
frame.add(scrollPane,BorderLayout.CENTER);
JPanel totalPanel=new JPanel();
totalPanel.setLayout(new GridLayout(1,3));
totalPanel.add(new JLabel("Total Bill:"));
totalField=new JTextField();
totalField.setEditable(false);
totalPanel.add(totalField);
JButton getBillButton=new JButton("Get Bill");
totalPanel.add(getBillButton);
frame.add(totalPanel,BorderLayout.SOUTH);

addButton.addActionListener(new ActionListener(){
public void actionPerformed(ActionEvent e){
addItemToOrder();
}
});

getBillButton.addActionListener(new ActionListener(){
public void actionPerformed(ActionEvent e){
showReceipt();
}
});

frame.setVisible(true);
}
private static void addItemToOrder(){
String name=nameField.getText();
double price=Double.parseDouble(priceField.getText());
int quantity=Integer.parseInt(quantityField.getText());
billingSystem.addOrderItem(name,price,quantity);
OrderItem orderItem=new OrderItem(new MenuItem(name,price),quantity);
tableModel.addRow(new Object[]{orderItem.getItem().getName(),orderItem.getItem().getPrice(),orderItem.getQuantity(),orderItem.getTotalPrice()});
totalField.setText(String.format("%.2f",billingSystem.calculateTotalBill()));
nameField.setText("");
priceField.setText("");
quantityField.setText("");
}
private static void showReceipt(){
StringBuilder receipt=new StringBuilder("Receipt:\n");
for (OrderItem orderItem : billingSystem.getOrderedItems()){
receipt.append(orderItem.getQuantity()).append(" x ");
receipt.append(orderItem.getItem().getName()).append(" @  ₹");
receipt.append(orderItem.getItem().getPrice()).append(" each= ₹");
receipt.append(orderItem.getTotalPrice()).append("\n");
}
receipt.append("\nTotal:  ₹").append(String.format("%.2f",billingSystem.calculateTotalBill()));
JOptionPane.showMessageDialog(frame,receipt.toString(),"Receipt",JOptionPane.INFORMATION_MESSAGE);
}
}
